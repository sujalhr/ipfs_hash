    <h1>Upload File to IPFS</h1>

    <div id="uploadContainer">
        <form id="uploadForm" enctype="multipart/form-data">
            <label for="file">Select file:</label>
            <input type="file" id="fileInput" name="file" required>

            <button type="submit">Upload</button>
        </form>

        <div id="progressBarContainer">
            <div id="progressBar"></div>
        </div>

        <div id="message"></div>
        <div id="error"></div>
    </div>

    <script>
        const uploadForm = document.getElementById('uploadForm');
        const messageDiv = document.getElementById('message');
        const errorDiv = document.getElementById('error');
        const progressBar = document.getElementById('progressBar');
        const progressBarContainer = document.getElementById('progressBarContainer');

        uploadForm.addEventListener('submit', function (event) {
            event.preventDefault();

            const fileInput = document.getElementById('fileInput');
            const file = fileInput.files[0];

            if (!file) {
                errorDiv.textContent = 'Please select a file to upload.';
                return;
            }

            const formData = new FormData();
            formData.append('file', file);

            // Reset progress bar
            progressBar.style.width = '0%';
            progressBarContainer.style.display = 'block';
            errorDiv.textContent = '';
            messageDiv.textContent = '';

            fetch('/upload', {
                method: 'POST',
                body: formData
            })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        messageDiv.innerHTML = `File uploaded successfully! <br>IPFS Hash: <a class="ipfs-link" href="https://gateway.pinata.cloud/ipfs/${data.hash}" target="_blank">${data.hash}</a>`;
                        progressBar.style.width = '100%';
                    } else {
                        errorDiv.textContent = data.message;
                        progressBarContainer.style.display = 'none';
                    }
                })
                .catch(err => {
                    errorDiv.textContent = 'Error uploading file.';
                    progressBarContainer.style.display = 'none';
                });
        });
    </script>
