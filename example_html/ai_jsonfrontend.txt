<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Upload File to IPFS</title>
</head>
<body>

    <h1>Upload File to IPFS</h1>

    <form id="uploadForm" enctype="multipart/form-data">
        <label for="file">Select file:</label>
        <input type="file" id="fileInput" name="file" required>
        <button type="submit">Upload</button>
    </form>

    <pre id="jsonResponse"></pre>

    <script>
        const uploadForm = document.getElementById('uploadForm');
        const jsonResponse = document.getElementById('jsonResponse');

        uploadForm.addEventListener('submit', function (event) {
            event.preventDefault();

            const fileInput = document.getElementById('fileInput');
            const file = fileInput.files[0];

            if (!file) {
                jsonResponse.textContent = JSON.stringify({ success: false, message: 'No file selected for uploading' }, null, 2);
                return;
            }

            const formData = new FormData();
            formData.append('file', file);

            fetch('/upload', {
                method: 'POST',
                body: formData
            })
            .then(response => response.json())
            .then(data => {
                // Display the JSON response containing the IPFS hash
                jsonResponse.textContent = JSON.stringify(data, null, 2);
            })
            .catch(err => {
                jsonResponse.textContent = JSON.stringify({ success: false, message: 'Error uploading file' }, null, 2);
            });
        });
    </script>

</body>
</html>
